CREATE PROCEDURE get_student_grades(IN p_reg_no CHAR(8))
BEGIN
    SELECT reg_no, c_code, grade
    FROM final_grades

END $$

CREATE PROCEDURE get_course_grades(IN p_c_code VARCHAR(10))
